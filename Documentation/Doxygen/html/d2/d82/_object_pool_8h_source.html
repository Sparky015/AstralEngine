175 inline void PushFreePointer(ElementType* elementPtr)
176 {
177 ASSERT(m_FreeCount != NumberOfElements, "Free has been called more times then Allocate!")
178 m_FreeCount++;
179 m_FreeArray[m_FreeCount - 1] = elementPtr;

185 inline ElementType* PopFreePointer()
186 {
187 ElementType* freeElement = m_FreeArray[m_FreeCount - 1];
